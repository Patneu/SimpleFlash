300    /**
301     * Handles clicks that are performed on the various 'xxxLink' {@link
302     * TextView TextViews} of this Activity. <br/><br/>
303     * 
304     * If one of the 'xxxLink' TextViews is clicked, this method tells the
305     * {@link #aboutScrollView} to scroll to the beginning of the corresponding
306     * 'xxxLabel' TextView. Therewith, this method provides the functionality
307     * of a table of contents for the user.
308     * 
309     * @param view the View that was clicked by the user
310     * 
311     * @since Version 1.0
312     */
313    public void onLinkClicked(View view)
314    {
315        switch(view.getId())
316        {
317            case R.id.app_description_link:
318                aboutScrollView.smoothScrollTo(0, (int) appDescriptionLabel.getY());
319                break;
320            case R.id.app_features_link:
321                aboutScrollView.smoothScrollTo(0, (int) appFeaturesLabel.getY());
322                break;
323            case R.id.app_permissions_link:
324                aboutScrollView.smoothScrollTo(0, (int) appPermissionsLabel.getY());
325                break;
326            case R.id.source_code_link:
327                aboutScrollView.smoothScrollTo(0, (int) sourceCodeLabel.getY());
328                break;
329            case R.id.app_license_link:
330                aboutScrollView.smoothScrollTo(0, (int) appLicenseLabel.getY());
331                break;
332            case R.id.liability_disclaimer_link:
333                aboutScrollView.smoothScrollTo(0, (int) liabilityDisclaimerLabel.getY());
334                break;
335            case R.id.additional_licenses_link:
336                aboutScrollView.smoothScrollTo(0, (int) additionalLicensesLabel.getY());
337                break;
338            case R.id.third_party_software_link:
339                aboutScrollView.smoothScrollTo(0, (int) thirdPartySoftwareLabel.getY());
340        }
341    }
342    
343    /**
344     * Handles the selection of items in the options menu. <br/><br/>
345     * 
346     * {@link ActionBarActivity#finish() Finishes} the Activity and {@link
347     * ActionBarActivity#setResult(int) sets} the {@link
348     * SimpleFlash#RESULT_FINISHED 'RESULT_FINISHED'} result (to finish all
349     * parent Activities as well) if the {@link #exitItem} was selected.
350     * 
351     * @param item the {@link MenuItem} selected by the user
352     * 
353     * @return true if the selected item was handled in any way, the return
354     *         value of the corresponding method of the super class otherwise
355     * 
356     * @see ActionBarActivity#onOptionsItemSelected(android.view.MenuItem)
357     * 
358     * @since Version 1.0
359     */
360    @Override
361    public boolean onOptionsItemSelected(MenuItem item)
362    {
363        switch(item.getItemId())
364        {
365            case R.id.exit_item:
366                setResult(SimpleFlash.RESULT_FINISHED);
367                finish();
368                return true;
369            default:
370                return super.onOptionsItemSelected(item);
371        }
372    }
