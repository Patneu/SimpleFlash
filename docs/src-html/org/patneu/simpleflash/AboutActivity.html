266    /**
267     * Handles clicks that are performed on the various 'xxxLink' {@link
268     * TextView TextViews} of this Activity. <br/><br/>
269     * 
270     * If one of the 'xxxLink' TextViews is clicked, this method tells the
271     * {@link #aboutScrollView} to scroll to the beginning of the corresponding
272     * 'xxxLabel' TextView. Therewith, this method provides the functionality
273     * of a table of contents for the user.
274     * 
275     * @param view the View that was clicked by the user
276     * 
277     * @since Version 1.0
278     */
279    public void onLinkClicked(View view)
280    {
281        switch(view.getId())
282        {
283            case R.id.app_description_link:
284                aboutScrollView.smoothScrollTo(0, (int) appDescriptionLabel.getY());
285                break;
286            case R.id.app_features_link:
287                aboutScrollView.smoothScrollTo(0, (int) appFeaturesLabel.getY());
288                break;
289            case R.id.app_permissions_link:
290                aboutScrollView.smoothScrollTo(0, (int) appPermissionsLabel.getY());
291                break;
292            case R.id.source_code_link:
293                aboutScrollView.smoothScrollTo(0, (int) sourceCodeLabel.getY());
294                break;
295            case R.id.app_license_link:
296                aboutScrollView.smoothScrollTo(0, (int) appLicenseLabel.getY());
297                break;
298            case R.id.liability_disclaimer_link:
299                aboutScrollView.smoothScrollTo(0, (int) liabilityDisclaimerLabel.getY());
300                break;
301            case R.id.additional_licenses_link:
302                aboutScrollView.smoothScrollTo(0, (int) additionalLicensesLabel.getY());
303                break;
304            case R.id.third_party_software_link:
305                aboutScrollView.smoothScrollTo(0, (int) thirdPartySoftwareLabel.getY());
306        }
307    }

309    /**
310     * Handles the selection of items in the options menu. <br/><br/>
311     * 
312     * {@link ActionBarActivity#finish() Finishes} the Activity and {@link
313     * ActionBarActivity#setResult(int) sets} the {@link
314     * SimpleFlash#RESULT_FINISHED 'RESULT_FINISHED'} result (to finish all
315     * parent Activities as well) if the {@link #exitItem} was selected.
316     * 
317     * @param item the {@link MenuItem} selected by the user
318     * 
319     * @return true if the selected item was handled in any way, the return
320     *         value of the corresponding method of the super class otherwise
321     * 
322     * @see ActionBarActivity#onOptionsItemSelected(android.view.MenuItem)
323     * 
324     * @since Version 1.0
325     */
326    @Override
327    public boolean onOptionsItemSelected(MenuItem item)
328    {
329        switch(item.getItemId())
330        {
331            case R.id.exit_item:
332                setResult(SimpleFlash.RESULT_FINISHED);
333                finish();
334                return true;
335            default:
336                return super.onOptionsItemSelected(item);
337        }
338    }
