053public class CameraOnNotification
054{
055    /**
056     * The id to {@link SimpleFlash.NotifyManager#sendNotification(int,
057     * android.app.Notification) send} a CameraOnNotification with. <br/><br/>
058     * 
059     * @since Version 1.0
060     */
061    public static final int CAMERA_ON_ID = 1;
062    
063    /**
064     * A value indicating to resume its sending Activity when a
065     * CameraOnNotification is clicked and/or cancelled. <br/><br/>
066     * 
067     * Primarily used as the 'click' or 'cancel' parameter of the {@link
068     * #getInstance(android.content.Context, android.content.Intent, int, int)}
069     * method.
070     * 
071     * @see #NOTIFICATION_FINISH
072     * @see #NOTIFICATION_NOTHING
073     * 
074     * @since Version 1.0
075     */
076    public static final int NOTIFICATION_RESUME = 0;
077    /**
078     * A value indicating to finish its sending Activity when a
079     * CameraOnNotification is clicked and/or cancelled. <br/><br/>
080     * 
081     * Primarily used as the 'click' or 'cancel' parameter of the {@link
082     * #getInstance(android.content.Context, android.content.Intent, int, int)}
083     * method.
084     * 
085     * @see #NOTIFICATION_RESUME
086     * @see #NOTIFICATION_NOTHING
087     * 
088     * @since Version 1.0
089     */
090    public static final int NOTIFICATION_FINISH = 1;
091    /**
092     * A value indicating to do nothing when a CameraOnNotification is clicked
093     * or cancelled. <br/><br/>
094     * 
095     * Primarily used as the 'click' or 'cancel' parameter of the {@link
096     * #getInstance(android.content.Context, android.content.Intent, int, int)}
097     * method.
098     * 
099     * @see #NOTIFICATION_RESUME
100     * @see #NOTIFICATION_FINISH
101     * 
102     * @since Version 1.0
103     */
104    public static final int NOTIFICATION_NOTHING = 2;
105    
106    /**
107     * A value to use as a parameter for {@link
108     * Intent#setAction(java.lang.String), indicating the receiving Activity to
109     * finish. <br/><br/>
110     * 
111     * @since Version 1.0
112     */
113    public static final String ACTION_FINISH = "android.intent.action.FINISH";
114    
115    private CameraOnNotification()
116    {
117    }
118    
119    /**
120     * Returns an instance of this application's standard {@link Notification},
121     * referred to as the 'CameraOnNotification'. <br/><br/>
122     * 
123     * This method gets a Notification to tell the user that the camera's
124     * flashlight is still turned on, since unintentionally leaving it on may
125     * damage the user's device. <br/><br/>
126     * 
127     * <b>Note:</b> Although this method is named 'getInstance', it doesn't
128     *              actually use any constructor of this class (which wouldn't
129     *              have sense anyway, since the class doesn't extend the
130     *              Notification class. Instead, it internally builds a new
131     *              Notification using the {@link NotificationCompat.Builder}
132     *              class.
133     * 
134     * @param context the {@link Context} of the Activity that wants to {@link
135     *                SimpleFlash.NotifyManager#sendNotification(int,
136     *                android.app.Notification) send} this Notification
137     * @param intent should be the {@link Intent} that originally started the
138     *               sending Activity, as provided by {@link
139     *               Activity#getIntent()}
140     * @param click defines what to do if the user clicks this Notification,
141     *              should be one of the static 'NOTIFICATION_XXX' values
142     * @param cancel defines what to do if the user cancels this Notification,
143     *               should be one of the static 'NOTIFICATION_XXX' values
144     * 
145     * @return an instance of this application's standard Notification with the
146     *         specified behavior
147     * 
148     * @since Version 1.0
149     */
150    public static Notification getInstance(Context context, Intent intent, int click, int cancel)
151    {
152        NotificationCompat.Builder nBuilder = new NotificationCompat.Builder(context);
153        nBuilder.setSmallIcon(R.drawable.ic_stat_notify);
154        nBuilder.setContentTitle(context.getString(R.string.app_name));
155        nBuilder.setContentText(context.getString(R.string.camera_on_notification_text));
156        nBuilder.setTicker(context.getString(R.string.camera_on_notification_text));
157        nBuilder.setPriority(NotificationCompat.PRIORITY_LOW);
158        
159        PendingIntent contentPending = null;
160        PendingIntent deletePending = null;
161        if(click != NOTIFICATION_NOTHING)
162        {
163            if(click == NOTIFICATION_RESUME)
164            {
165                contentPending = PendingIntent.getActivity(context, 0, intent, PendingIntent.FLAG_UPDATE_CURRENT);
166            }
167            if(click == NOTIFICATION_FINISH)
168            {
169                Intent finishIntent = intent.cloneFilter();
170                finishIntent.setAction(ACTION_FINISH);
171                contentPending = PendingIntent.getActivity(context, 0, finishIntent, PendingIntent.FLAG_UPDATE_CURRENT);
172            }
173            nBuilder.setContentIntent(contentPending);
174        }
175        if(cancel != NOTIFICATION_NOTHING)
176        {
177            if(cancel == NOTIFICATION_RESUME)
178            {
179                deletePending = PendingIntent.getActivity(context, 0, intent, PendingIntent.FLAG_UPDATE_CURRENT);
180            }
181            if(cancel == NOTIFICATION_FINISH)
182            {
183                Intent finishIntent = intent.cloneFilter();
184                finishIntent.setAction(ACTION_FINISH);
185                deletePending = PendingIntent.getActivity(context, 0, finishIntent, PendingIntent.FLAG_UPDATE_CURRENT);
186            }
187            nBuilder.setDeleteIntent(deletePending);
188        }
189        
190        return nBuilder.build();
191    }
