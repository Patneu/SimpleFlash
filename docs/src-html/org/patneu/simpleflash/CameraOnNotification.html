087public class CameraOnNotification
088{
089    /**
090     * The id to {@link SimpleFlash.NotifyManager#sendNotification(int,
091     * android.app.Notification) send} a CameraOnNotification with. <br/><br/>
092     * 
093     * @since Version 1.0
094     */
095    public static final int CAMERA_ON_ID = 1;
096    
097    /**
098     * A value indicating to resume its sending Activity when a
099     * CameraOnNotification is clicked and/or cancelled. <br/><br/>
100     * 
101     * Primarily used as the 'click' or 'cancel' parameter of the {@link
102     * #getInstance(android.content.Context, android.content.Intent, int, int)}
103     * method.
104     * 
105     * @see #NOTIFICATION_FINISH
106     * @see #NOTIFICATION_NOTHING
107     * 
108     * @since Version 1.0
109     */
110    public static final int NOTIFICATION_RESUME = 0;
111    /**
112     * A value indicating to finish its sending Activity when a
113     * CameraOnNotification is clicked and/or cancelled. <br/><br/>
114     * 
115     * Primarily used as the 'click' or 'cancel' parameter of the {@link
116     * #getInstance(android.content.Context, android.content.Intent, int, int)}
117     * method.
118     * 
119     * @see #NOTIFICATION_RESUME
120     * @see #NOTIFICATION_NOTHING
121     * 
122     * @since Version 1.0
123     */
124    public static final int NOTIFICATION_FINISH = 1;
125    /**
126     * A value indicating to do nothing when a CameraOnNotification is clicked
127     * or cancelled. <br/><br/>
128     * 
129     * Primarily used as the 'click' or 'cancel' parameter of the {@link
130     * #getInstance(android.content.Context, android.content.Intent, int, int)}
131     * method.
132     * 
133     * @see #NOTIFICATION_RESUME
134     * @see #NOTIFICATION_FINISH
135     * 
136     * @since Version 1.0
137     */
138    public static final int NOTIFICATION_NOTHING = 2;
139    
140    /**
141     * A value to use as a parameter for {@link
142     * Intent#setAction(java.lang.String), indicating the receiving Activity to
143     * finish. <br/><br/>
144     * 
145     * @since Version 1.0
146     */
147    public static final String ACTION_FINISH = "android.intent.action.FINISH";
148    
149    private CameraOnNotification()
150    {
151    }
152    
153    /**
154     * Returns an instance of this application's standard {@link Notification},
155     * referred to as the 'CameraOnNotification'. <br/><br/>
156     * 
157     * This method gets a Notification to tell the user that the camera's
158     * flashlight is still turned on, since unintentionally leaving it on may
159     * damage the user's device. <br/><br/>
160     * 
161     * <b>Note:</b> Although this method is named 'getInstance', it doesn't
162     *              actually use any constructor of this class (which wouldn't
163     *              have sense anyway, since the class doesn't extend the
164     *              Notification class. Instead, it internally builds a new
165     *              Notification using the {@link NotificationCompat.Builder}
166     *              class.
167     * 
168     * @param context the {@link Context} of the Activity that wants to {@link
169     *                SimpleFlash.NotifyManager#sendNotification(int,
170     *                android.app.Notification) send} this Notification
171     * @param intent should be the {@link Intent} that originally started the
172     *               sending Activity, as provided by {@link
173     *               Activity#getIntent()}
174     * @param click defines what to do if the user clicks this Notification,
175     *              should be one of the static 'NOTIFICATION_XXX' values
176     * @param cancel defines what to do if the user cancels this Notification,
177     *               should be one of the static 'NOTIFICATION_XXX' values
178     * 
179     * @return an instance of this application's standard Notification with the
180     *         specified behavior
181     * 
182     * @since Version 1.0
183     */
184    public static Notification getInstance(Context context, Intent intent, int click, int cancel)
185    {
186        NotificationCompat.Builder nBuilder = new NotificationCompat.Builder(context);
187        nBuilder.setSmallIcon(R.drawable.ic_stat_notify);
188        nBuilder.setContentTitle(context.getString(R.string.app_name));
189        nBuilder.setContentText(context.getString(R.string.camera_on_notification_text));
190        nBuilder.setTicker(context.getString(R.string.camera_on_notification_text));
191        nBuilder.setPriority(NotificationCompat.PRIORITY_LOW);
192        
193        PendingIntent contentPending = null;
194        PendingIntent deletePending = null;
195        if(click != NOTIFICATION_NOTHING)
196        {
197            if(click == NOTIFICATION_RESUME)
198            {
199                contentPending = PendingIntent.getActivity(context, 0, intent, PendingIntent.FLAG_UPDATE_CURRENT);
200            }
201            if(click == NOTIFICATION_FINISH)
202            {
203                Intent finishIntent = intent.cloneFilter();
204                finishIntent.setAction(ACTION_FINISH);
205                contentPending = PendingIntent.getActivity(context, 0, finishIntent, PendingIntent.FLAG_UPDATE_CURRENT);
206            }
207            nBuilder.setContentIntent(contentPending);
208        }
209        if(cancel != NOTIFICATION_NOTHING)
210        {
211            if(cancel == NOTIFICATION_RESUME)
212            {
213                deletePending = PendingIntent.getActivity(context, 0, intent, PendingIntent.FLAG_UPDATE_CURRENT);
214            }
215            if(cancel == NOTIFICATION_FINISH)
216            {
217                Intent finishIntent = intent.cloneFilter();
218                finishIntent.setAction(ACTION_FINISH);
219                deletePending = PendingIntent.getActivity(context, 0, finishIntent, PendingIntent.FLAG_UPDATE_CURRENT);
220            }
221            nBuilder.setDeleteIntent(deletePending);
222        }
223        
224        return nBuilder.build();
225    }
